// 0. Lưu ý quan trọng: Mở terminal ở đúng thư mục chứa file scraper.js
// Chạy lệnh sau để chuyển đúng vào thư mục nếu gặp lỗi "Script not found":
// cd "C:\Users\IOC\Desktop\Kiem com\Cong-Thong-Tin-Quoc-Gia-main\Cong-Thong-Tin-Quoc-Gia-main"

// 1. Chạy ngầm và cào liên tục (Tự động khởi động lại nếu lỗi)
pm2 start scraper.js --name "dvc-scraper"

// 2. Chạy 1 lần không tự khởi động lại (Chạy xong tự tắt tiến trình)
pm2 start scraper.js --name "dvc-scraper" --no-autorestart

// 3. Xoá tiến trình cũ và chạy lại 1 lần
pm2 delete dvc-scraper && pm2 start scraper.js --name "dvc-scraper" --no-autorestart

// 4. Chạy định kỳ 1 tuần 1 lần (Ví dụ: chạy vào lúc 1h sáng Chủ Nhật hàng tuần)
pm2 start scraper.js --name "dvc-scraper" --cron "0 1 * * 0" --no-autorestart

// 4b. Chạy vào lúc 0h00 sáng thứ 2 hàng tuần (Đầu tuần)
pm2 delete dvc-scraper
pm2 start scraper.js --name "dvc-scraper" --cron "0 0 * * 1" --no-autorestart
pm2 save --force

// 5. Xem log (theo dõi tiến trình cào dữ liệu)
pm2 logs dvc-scraper --lines 100

// 6. Dừng cào dữ liệu
pm2 stop dvc-scraper

// 7. Khởi động lại
pm2 restart dvc-scraper

// 8. Xoá tiến trình khỏi PM2
pm2 delete dvc-scraper

// 9. Lưu trạng thái PM2 (để tự bật lại khi khởi động máy tính)
pm2 save --force

// 10. Tắt sạch mọi thứ (Kill PM2)
pm2 kill
